  def initialize(*args, &block)
    self.class.send :init_exit_handler
    @thread = Thread.new(*args) do |*targs|
      begin
        block.call(*targs)
      rescue => e
        @exception = e
        self.class.send :handle_exception, e
      end
    end
    @thread[:real_ultimate_power] = true
  end
